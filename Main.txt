import java.io.*;
import java.util.*;

class Main {
    public static void main(String[] args) throws IOException {
        File num = new File("C:\\Users\\Student\\IdeaProjects\\task4\\src\\umbers.txt");
        int[] arn = new int[6];
        FileReader inputFil = new FileReader(num);
        BufferedReader in = new BufferedReader(inputFil);
        String s = in.readLine();
        int i = 0;
        while (s != null) {
            s = s.trim();
            if (s.length() == 0) {
                continue;
            }
            arn[i] = Integer.parseInt(s);
            System.out.println(arn[i]);
            s = in.readLine();
            i++;
        }
        in.close();
    }
}
